% =========================
%         COMMANDS
% =========================

% Check if a variable is defined
\newcommand{\checkvardefined}[1]{
	\ifthenelse{\isundefined{#1}}{
		\errmessage{LaTeX Warning: Variable \noexpand#1 not defined}
		\stop}{
	}
}
% Set the geometry of the page (requires GEOMETRY)
\newcommand{\setpagemargincm}[4]{
    \newgeometry{left=#1cm, top=#2cm, right=#3cm, bottom=#4cm}
}

% Special table columns (requires ARRAY)
\newcolumntype{C}[1]{>{\centering\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}
\newcolumntype{L}[1]{>{\raggedright\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}
\newcolumntype{P}[1]{>{\centering\arraybackslash}p{#1}}
\newcolumntype{R}[1]{>{\raggedleft\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}

% Command for default float location
\makeatletter
    \providecommand*\setfloatlocation[2]{\@namedef{fps@#1}{#2}}
    \g@addto@macro\@floatboxreset\centering
\makeatother

% DEFINE TABLENOTE COMMAND
\newcommand{\tablenote}[2][0pt]{
    \vspace{#1}\\{\footnotesize#2\par}
    }
    
% CREATE APPENDIX
\newcommand{\beginappendix}[0]{
    \clearpage
    \appendix
    \phantomsection
    \addcontentsline{toc}{section}{Appendices}
    \titleformat{\section}{\normalfont\LARGE\bfseries\raggedright}{Appendix~\thesection.}{2ex}{}
    \setcounter{table}{0}
    \setcounter{figure}{0}
    \renewcommand{\thetable}{\Alph{section}\arabic{table}}
    \renewcommand{\thefigure}{\Alph{section}\arabic{figure}}
}
    
% DEFINE MATH FORMULAS
\DeclareMathOperator*{\argmax}{arg\,max}
\DeclareMathOperator*{\argmin}{arg\,min}

% CREATE DEDICATION ENVIRONMENT
\newenvironment{dedication}{
    \clearpage
    \thispagestyle{empty}
    \vspace*{\stretch{1}}
    \itshape
    \raggedleft
    }{
    \par
    \vspace{\stretch{3}}
    \clearpage
    }
